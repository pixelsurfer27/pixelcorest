# Text vectorization using Bag-of-words

# Source Code
import pandas as pd
from sklearn.feature_extraction.text import CountVectorizer

df = pd.DataFrame({
    'text': [
        "Cats are running quickly",
        "Boys are lacking loudly",
        "They study in university",
        "They are running to reach the target"
    ]
})

# Original DataFrame!!
print('Original DataFrame!!')
print(df)

vectorizer = CountVectorizer()
vectors = vectorizer.fit_transform(df)
bow_df = pd.DataFrame(vectors.toarray(), columns=vectorizer.get_feature_names_out())
print('\nBag of words representation')
print(bow_df)

# Filling Missing values using Forward fill and Backward fill

# Source Code:
import pandas as pd
import numpy as np
from sklearn.impute import SimpleImputer

data = {
    'Name': ['Ram', 'Sudar', np.nan, 'Ankit', 'Ritu', 'Gulam', np.nan, 'Rita'],
    'Age': [64, 26, 35, np.nan, 38, 42, np.nan, 42],
    'Height': [252, np.nan, 135, 142, np.nan, 157, 161, 146],
    'Weight': [68, np.nan, 75, 70, 83, 57, 43, np.nan],
    'Grade': [np.nan, 'B', np.nan, 'D', 'C', 'A', np.nan, 'B']
}

df = pd.DataFrame(data)
print("Original Database")
print(df)

df_f = df.ffill()
df_b = df.bfill()

print("\nAfter Forward fill")
print(df_f)

print("\nAfter Backward fill")
print(df_b)

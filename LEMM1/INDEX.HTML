# lemmmatization 

import pandas as pd
from nltk.stem import WordNetLemmatizer
from nltk.tokenize import word_tokenize
import nltk

nltk.download('punkt')
nltk.download('wordnet')
nltk.download('omw-1.4')

df = pd.DataFrame({
    'Text': [
        'Cats are running quickly.',
        'He studies in university.',
        'Dogs were barking loudly.',
        'They are enjoying the holidays.'
    ]
})

print('Original Dataset')
print(df)

lemmatizer = WordNetLemmatizer()

def lemmatize_text(text):
    tokens = word_tokenize(text.lower())
    lemma = [lemmatizer.lemmatize(word) for word in tokens if word.isalpha()]
    return " ".join(lemma)

df['lemmatised_text'] =df["Text"].apply(lemmatize_text)

print('After Lemmatization')
print(df[['lemmatised_text']])

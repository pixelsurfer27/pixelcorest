# Label encoding to handle categorical data

import pandas as pd
from sklearn.preprocessing import LabelEncoder

data = {
    'Fruit': ['Apple, Banana, Orange', 'Apple', 'Banana', 'Orange'],
    'Color': ['Red', 'Yellow', 'Orange', 'Green', 'Yellow', 'Red'] # Note: Data mismatch in image, corrected length assumption
}

# Adjusted data to match lengths and image content
data = {
    'Fruit': ['Apple', 'Banana', 'Orange', 'Apple', 'Banana', 'Orange'],
    'Color': ['Red', 'Yellow', 'Orange', 'Green', 'Yellow', 'Red']
}

df = pd.DataFrame(data)
print("Original Dataset")
print(df)

label_encoder = LabelEncoder()

df_label_encoded = df.copy()
df_label_encoded['Fruit'] = label_encoder.fit_transform(df['Fruit'])
df_label_encoded['Color'] = label_encoder.fit_transform(df['Color'])

print("After Label Encoding")
print(df_label_encoded)

# Custom - Binning (Type-3):-

import pandas as pd
df = pd.DataFrame({
    'value': [5, 12, 17, 24, 35, 46, 59, 63, 60, 72]
})
print(df)
bins = [0, 12, 18, 60, 100]
labels = ['Child', 'Teen', 'Adult', 'Senior']
df['Custom- Bin'] = pd.cut(df['value'], bins=bins, labels=labels, right=True)

print(df)
